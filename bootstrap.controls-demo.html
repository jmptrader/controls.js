Buttons, size and styles

example.add("btn_prev:Button#size=2,style=success",  {$icon:"chevron-left"})
.listen("click", function()
{
    if (btn_size > 0)
        btn_size--;

    this.parent.forEach(function(button) { button.size(btn_size); });
});

Forms basic example

Example block-level help text here.

$$C("bootstrap.Form");
$$C("bootstrap.FormGroup", function(group)
{
    $C("label", {$text:"Email address", for:"exampleInputEmail1"});
    $C("input:input", {type:"email", class:"form-control", id:"exampleInputEmail1", placeholder:"Enter email"});
    
    // e-mail input validation:
    var regexp = /(?:[a-z0-9!#$%&.../;
    group.input.listen("keyup", group.input, function()
    {
        if (this._element)
        {
            if (regexp.test(this._element.value))
                group.class("", "has-error");
            else
                group.class("has-error");
        }
    });
});
